   subroutine darrays_init(comm)
      type(comm_t), intent(in) :: comm
      integer(int32) :: i

      global_comm = comm%duplicate()
      do i = 1, MAX_ARRAYS
         registry(i)%active = .false.
         registry(i)%handle = -1
      end do
      initialized = .true.
   end subroutine darrays_init
