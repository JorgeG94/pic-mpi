   subroutine darrays_finalize()
      integer(int32) :: i

      if (.not. initialized) return

      do i = 1, MAX_ARRAYS
         if (registry(i)%active) then
            call darray_destroy(i)
         end if
      end do

      call global_comm%finalize()
      initialized = .false.
   end subroutine darrays_finalize
