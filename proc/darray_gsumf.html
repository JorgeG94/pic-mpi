   subroutine darray_gsumf(buffer, n)
      real(dp), intent(inout) :: buffer(*)
      integer(int32), intent(in) :: n
      real(dp), allocatable :: temp(:)

      allocate (temp(n))
      temp = buffer(1:n)
      call allreduce(wrapper_comm, temp, n)
      buffer(1:n) = temp
      deallocate (temp)
   end subroutine darray_gsumf
